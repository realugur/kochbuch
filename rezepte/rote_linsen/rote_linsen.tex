\begin{recipe}{Kırmızı Mercimek}

    \ingredient{Zwiebeln} 1-2
    \ingredient{Karotten} 1-2
    \ingredient{Kartoffeln} 1-2
    \ingredient{rote Linsen} ca. 2-3 Becher
    \ingredient{warmes Wasser} ca. 12 Becher 
    \ingredient{(Mehl)} 1-2 Löfel, nach Wunsch
    \ingredient{(scharfe Peperoni)} nach Wunsch
    \ingredient{Salz, Pfeffer}
    \ingredient{}
    \ingredient{}

    \preparationtime{$< 1$ Std.}
    \sidedish{Brötchen}
    \portions{ca. 5}
    \fourstar
    \ingredient{}
    \ingredient{}

    \ingredient{1.} die Linsen in einem ausreichend großen Behälter grundlich waschen und mehrmals spülen, bis das Wasser klar ist, das Wasser am jedes mal weg kippen 
    \ingredient{2.} das Gemüse schälen, waschen und grob zerkleinern
    \ingredient{3.} Öl in einem Topf erhitzen und die grob zerkleienerten Zwiebeln anbraten
    \ingredient{4.} (das Mehl hinzugeben und eine weitere Minute anbraten)
    \ingredient{5.} das restliche Gemüse hinzugeben und etwas umrühren 
    \ingredient{6.} Salz, Pfeffer und Linsen hinzugeben
    \ingredient{7.} Wasser hinzugeben, Deckel auf den Topf und eine halbe Stunde köcheln lassen
    \ingredient{8.} wenn das Gemüse weich gekocht ist, alles zusammen pürieren und weitere 5 min ziehen lassen 

\end{recipe}
