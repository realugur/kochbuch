\addcontentsline{toc}{section}{Bananenbrot}
\begin{recipe}{Bananenbrot}

    \ingredient{Mehl} 2 Tassen
    \ingredient{Backpulver} 1 Teelöffel
    \ingredient{Natron} 1 Teelöffel
    \ingredient{Vanillezucker} 1 Teelöffel
    \ingredient{Zimt} 1 Teelöffel 
    \ingredient{Nüsse und/oder Samen} gehackt, z.B. gehackte Wallnüsse und Leinsamen
    \ingredient{Zartbitter Schokolade} 1 Tafel, zerkleinert
    \ingredient{(andere Zutaten} z.B. gehackte Datteln)
    \ingredient{Bananen} 3-5
    \ingredient{Ahornsirup} 1-2 Löffel, evtl. sonst Honig
    \ingredient{Margarine / Öl}
    \ingredient{pflanzliche Milch} oder Wasser
    \ingredient{Salz} 1 Prise
    \ingredient{}
    \ingredient{}

    \preparationtime{ca. 1.5 Std.}
    \fivestar
    \ingredient{}
    \ingredient{}

    \ingredient{1.} Ofen auf 160\textdegree\ heizen
    \ingredient{2.} in einer Schüssel Mehl, Backpulver, Natron, Vanillezucker, Zimt und Salz vermischen
    \ingredient{3.} gemäßigte Menge Schokolade, Nüsse, Samen unterrühren (evtl. auch andere Zutaten dazu geben)
    \ingredient{4.} Bananen in einer anderen Schale mit einer Gabel zermatschen, Sirup und Margarine unterrühren
    \ingredient{5.} flüssige Zutaten, zum Mehl unterrühren, evtl mit Milch oder Wasser verdünnen
    \ingredient{6.} Backform mit Backpapier auslegen und Teig hineingeben
    \ingredient{7.} Teig mit toppings dekorieren (Schokolade, Nüsse, Früchte)
    \ingredient{8.} ca. 1 Std backen

\end{recipe}
