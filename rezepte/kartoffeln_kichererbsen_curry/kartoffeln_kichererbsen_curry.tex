\addcontentsline{toc}{section}{Kartoffeln, Kichererbsen Curry}
\begin{recipe}{Kartoffeln, Kichererbsen Curry}

    \ingredient{Reis} ca. 250 g
    \ingredient{Zwiebeln} 2
    \ingredient{Kartoffeln} ca. 4
    \ingredient{Karotten} 1 große, oder 2 kleine
    \ingredient{Currypaste} 2-3 Tl
    \ingredient{Currypulver} 1-2 Tl
    \ingredient{Kokosmilch} 500 ml
    \ingredient{Gemüsebrühe} 250 ml
    \ingredient{Kichererbsen} 500 g
    \ingredient{(gefrorene) Erbsen} kleine Dose, ca. 300 g
    \ingredient{Salz, Pfeffer}
    \ingredient{}
    \ingredient{}

    \sidedish{Reis}
    \preparationtime{$< 1$ Std.}
    \portions{ca. 5 }
    \fourstar
    \ingredient{}
    \ingredient{}

    \ingredient{1.} den Reis wie gewohnt zubereiten
    \ingredient{2.} die Zwiebeln in feine Streifen schneiden
    \ingredient{3.} Karotten in dünne Scheiben schneiden
    \ingredient{4.} Kartoffeln schälen und würfeln
    \ingredient{5.} Öl in einer Pfanne erhitzen und zwiebeln ca. 2 min glasig anbraten
    \ingredient{6.} Kartoffeln hinzugeben und ein wenig weiter anbraten
    \ingredient{7.} Curry und Currypaste unterrühren und eine Minute ziehen lassen
    \ingredient{5.} restliche Zutaten hinzufügen und 20 min köcheln lassen

\end{recipe}
