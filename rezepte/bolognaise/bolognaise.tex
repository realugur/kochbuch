\addcontentsline{toc}{section}{Bolognaise}
\begin{recipe}{Bolognaise}

    \ingredient{Zwiebeln} 2, gewürfelt
    \ingredient{Chillieschote} 1, klein geschnitten
    \ingredient{Knoblauch} 1-2, gehackt
    \ingredient{Karotten} 2-3
    \ingredient{Paprika} 1-2
    \ingredient{Linsen} grüne oder rote
    \ingredient{Tomatenmark} 1-2 Löfel
    \ingredient{Tomaten} 2 Dosen, gehackt oder pürriert
    \ingredient{Gewürze} Salz, Pfeffer, \dots
    \ingredient{(Sojagranulat)} 
    \ingredient{(Sirup)} oder 3 Zuckerwürfel
    \ingredient{}
    \ingredient{}

    \sidedish{Nudeln}
    \preparationtime{ca. 1 Std.}
    \portions{ca. 3 Tage}
    \fivestar
    \ingredient{}
    \ingredient{}

    \ingredient{1.} Linsen gut waschen und in warmem Wasser ruhen lassen
    \ingredient{2.} Zwiebeln, Chillieschote, Knoblauch im Topf mit etwas Öl bei nicht zu hoher Temperatur lange anbraten
    \ingredient{3.} Karotten und Paprika im Mixer ganz fein pürrieren und in den Topf geben + Tomatenmark in den Topf
    \ingredient{4.} Linsen abspülen und in den Topf
    \ingredient{5.} Tomaten aus der Dose in den Topf, Dosen zur Hälfte mit Wasser auffüllen und in den Topf kippen
    \ingredient{6.} evtl. etwas Soja-Granulat direkt drüber streuen und umrühren
    \ingredient{7.} würzen und nach Geschmack Sirup oder etwas Zucker rein

\end{recipe}
