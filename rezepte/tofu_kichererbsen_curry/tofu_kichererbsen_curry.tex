\addcontentsline{toc}{section}{Tofu, Kichererbsen Curry}
\begin{recipe}{Tofu, Kichererbsen Curry}

    \ingredient{Zwiebeln} 2, gewürfelt
    \ingredient{Tofu} 1 Block, klein gewürfelt
    \ingredient{Knoblauch} 1-2 Zehen, gehackt
    \ingredient{Chillieschote} 1, klein geschnitten
    \ingredient{Karotten} 2-3, gewürfelt
    \ingredient{Paprika} 1-2
    \ingredient{Linsen} grüne oder rote
    \ingredient{Kicherebsen} 1 Dose 
    \ingredient{Tomaten} 2 Dosen, gehackt und pürriert 
    \ingredient{Tomatenmark} 1-2 Löfel
    \ingredient{Gewürze} Salz, Pfeffer, \dots
    \ingredient{Currypaste} 1-2 Teelöfel 
    \ingredient{}

    \sidedish{Reis}
    \preparationtime{ca. 1-2 Std.}
    \portions{ca. 3 Tage}
    \fivestar
    \ingredient{}

    \ingredient{1.} Linsen gut waschen und in warmem Wasser ruhen lassen 
    \ingredient{2.} Reis kochen 
    \ingredient{3.} Zwiebeln, Chillieschote, Tofu, Knoblauch im Topf mit etwas Öl bei nicht zu hoher Temperatur lange anbraten
    \ingredient{4.} Karotten, Paprika, Curry, Gewürze, Kichererbsen, Tomatenmark in den Topf
    \ingredient{5.} Linsen abspülen und in den Topf
    \ingredient{6.} Tomaten aus der Dose in den Top
    \ingredient{7.} köcheln lassen
\end{recipe}
